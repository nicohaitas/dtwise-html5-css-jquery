Quick setup — if you’ve done this kind of thing before
https:
	https://github.com/nicohaitas/dtwise-html5-css-jquery.git
ssh:
	git@github.com:nicohaitas/dtwise-html5-css-jquery.git

…or create a new repository on the command line

	echo "# dtwise-html5-css-jquery" >> README.md
	git init
	git add README.md
	git commit -m "first commit"
	git remote add origin git@github.com:nicohaitas/dtwise-html5-css-jquery.git
	git push -u origin master

…or push an existing repository from the command line

	git remote add origin git@github.com:nicohaitas/dtwise-html5-css-jquery.git
	git push -u origin master
